 /*===============================================*/
	var  getXYByDeg = function(iDeg,iRadius){
                if(iDeg==0)
                {
                    return {x:0,y:iRadius};
                }
                else if(iDeg==90)
                {
                    return {x:iRadius,y:0};
                }
                return {x:Math.sin(iDeg*Math.PI/180)*iRadius,y:Math.cos(iDeg*Math.PI/180)*iRadius};
            }
