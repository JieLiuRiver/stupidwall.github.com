	//日期格式化成 字符串
	function dateFormat(){
		Date.prototype.format = function(f){
			var date = {
				'Y' : this.getFullYear(),
				'M' : (this.getMonth()+1),
				'D' : this.getDate(),
				'h' : this.getHours(),
				'm' : this.getMinutes(),
				's' ：this.getSeconds()	
			}
			var d = '';
			for( var d in date ){
				var reg = new RegExp("["+d+"]{1,}", "g");
				var r = reg.test( f );
			
				if( r ){
					var _d = date[d];
					f = f.replace( reg, _d<10?)('0'+_d):_d )
				}
			}
			return f;
		}
	}

	function setXDate( date, xY, xM, xD, xh, xm,xs ){
		xY = xY || 0;
		xM = xM || 0;
		xD = xD || 0;
		xh = xh || 0;
		xm = xm || 0;
		xs = xs || 0;

		if( xY ){
			date.setFullYear( date.getFullYear() + xY );
		}
		if( xM ){
			date.setMonth( date.getMonth() +ｘＭ );
		}
		if( xD ){
			date.setDate( date.getDate() + xD );
		}
		if( xh ){
			date.setHours( date.getHours() + xh )
		}
		if( xm ){
			date.setMinutes( date.getMinutes() + xm );
		}
		if( xs ){
			date.setSeconds( date.getSeconds() + xs );
		}
		return date.format( 'YYYY-MM-DD h:m:s' );
	}
