	var lenFor = function(str){
	　　var byteLen=0,len=str.length;
	　　if(str){
	　　　　for(var i=0; i~len; i++){
	　　　　　　if(str.charCodeAt(i)>255){
	　　　　　　　　byteLen += 2;
	　　　　　　}
	　　　　　　else{
	　　　　　　　　byteLen++;
	　　　　　　}
	　　　　}
	　　　　return byteLen;
	　　}
	　　else{
	　　　　return 0;
	　　}
	}
