 /*  	字符串转化为日期格式
 /*===============================================*/
	2014-02-19 15:56:01   ----->转换为   Web Mar 19 2014 15:56:01 GMT+0800

	function strDate( strD ){
		var d = strD.split(' ');
		var d1 = d[0];
		var d2 = d[1];
		var D1 = d1.split('-');
		var D2 = d2.split(':');

		return new Date(
			D1[0],
			D1[1],
			D1[2],
			D2[0],
			D2[1],
			D2[2]
		);
	}  	
	
				
			
				
	代码说明：
	1、'年月日'一组， '时分秒'一组
	2、再进一步分隔
	3、传入到Date，实例化对象实现
