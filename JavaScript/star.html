		window.onload = function(){
			//获取指定类型的节点
			function getTypeElement(es, type){
				var len = es.length;
				var eArr = [];
				var esl = null;
				for( var i=0;i<len; i++ ){
					esl = es[i];
					if( esl.nodeName.replace('#','').toLocaleLowerCase() == type ){
						eArr.push(esl);
					}
				}
				return eArr;
			}

			//等级星投效果
			function vote( e ){
