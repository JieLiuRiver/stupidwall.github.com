	var baiduMap = null; 
	var intBaiduMap = function( m ){
		baiduMap = new BMap.Map(m);  //创建BMap实例
		var point = new BMap.Point(116.404, 39.915);   //创建点坐标
		baiduMap.centerAndZoom(point, 15);   //初始化地图，设置中心点坐标，地图级别
		baiduMap.enableScrollWheelZoom(); //启用滚轮放大缩小，默认禁止
		baiduMap.disableDragging();    // 禁用地图拖拽
	}
	intBaiduMap();

	1、引入js、加上申请的key
	2、实例化百度地图对象
	3、初始化地图对象的API
	4、根据需要，定制化地图模块
