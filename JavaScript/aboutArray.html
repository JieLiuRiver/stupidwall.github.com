				 /*  	关于数组的各个方法
				 /*===============================================*/
					1)、 希望在数组中搜索一个特定的值，找到就获取数组元素的索引
						var arr = ['a','b','c','d','a'];
						console.log( arr.indexOf( 'b' ) );  //打印出1 	
						console.log( arr.indexOf('a', 2) );  //打印出4

					2)、 删除或者替换数组元素
						var arr = ['a','b','c','d'];
						//删除
						arr.splice( arr.indexOf('b'), 1 );  // a c d 
						//替换
						arr.splice( arr.lastIndexOf('d'), 1, 'f' ); // a b c f

						//splice第一个参数负数
						arr.splice( -1, 1, 'h' );  //a b c h
						
						//删除后面
						arr.splice( 2 );  // a b

						//2个元素替代
						arr.splice( 2, 1, 'c1', 'c2' );   // a b c1 c2 d

						//循环使用splice
						var arr  = ['a1','b1','c1','a1','d1','a1'];
						while( arr.indexOf('a1') != -1 ){
							arr.splice( arr.indexOf('a1'), 1, '**' );
						}
						//  ['**', 'b1', 'c1','**','d1','**'] 

					3)、 slice方法，浅拷贝一部分
						var arr = ['a1','a2','a3','a4'];
						var new = arr.slice( 1,3 );
						console.log( new )  // [ 'a2', 'a3' ]

						这里复制的是数组里面字符串或者数字，是按照值来复制的，没什么问题。
						如果说，复制的里面的内容，也是一个对象，那就意味着，
						当我复制出来的新数组，改变其中的内容后，这个时候，旧数组也会体现出来。
						因为浅拷贝，拷贝的对象，指向的是相同的对象。

					4)、 使用forEach
						var arr = ['a1','a2','a3','a4','a1','a5','a1'];
						arr.forEach( function( element, index, array ){
							(element == 'a1') && ( array[index] = '**' );
						} );
					

					5)、 使用map()方法
						var arr = [23, 255, 122, 5, 16, 99];
						var newArr = arr.map(function( element ){
							return element.toString(16);
						})
						console.log(  newArr );  // ['17','ff','7a','5','10','63']

					6)、过滤数组 filter()方法，这个方法是ECMAScript5新增加的方法
						var arr  = ['**','bb','cd','**','cc','**'];
						var newArr = arr.filter( function(element){
							return ( element !== '**' );						
						} );
						console.log( newArr )  //['bb','cd','cc']

					7)、 every() 、 some()方法 验证数组的内容
						var arr1 = ['**','aaa','bbb'];
						var result = arr1.every( function( element,index,array ){
							var reg = /^[a-zA-Z]+$/;
							return reg.test( element );
						} );
						console.log( result ); // false
						//当全部满足条件的时候，才会返回true

						var result2 = arr1.some( function(element, index, array){
							var reg = /^[a-zA]+$/;
							return reg.test( element );
						} );
						console.log( result2 ); //true
						//只要有一个满足，就会返回truei
