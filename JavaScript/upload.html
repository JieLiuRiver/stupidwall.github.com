					 /*  				  	可拖拽文件的JQUERY文件上传控件美化插件
					 /*===============================================
							1、要引入的文件：
								1）、css/normalize.css
								2）、css/zTreeStyle/default.css
								3）、css/zTreeStyle/demo.css
								4）、css/zTreeStyle/dropify.min.css
								5）、jquery.min.js
								6）、js/dropify.js
					
						   	2、使用：
								html：
								<input type="file" id="input-file-now" class="dropify" data-default-file="" />
								<input type="file" id="input-file-events" class="dropify-event" data-default-file="src/images/test-image-1.jpg" />
								<input type="file" id="input-file-french-1" class="dropify-fr" data-default-file="" />

								js：
								//基本用法
								$('.dropify').dropify();
								
								//添加信息：
								$('.dropify-fr').dropify({
							                    messages: {
							                        'default': '点击或拖拽文件到这里',
										        'replace': '点击或拖拽文件到这里来替换文件',
										        'remove':  '移除文件',
										        'error':   '对不起，你上传的文件太大了'
							                    }
							                });

					              //监听移除事件
					              var drEvent = $('.dropify-event').dropify();

					                drEvent.on('dropify.beforeClear', function(event, element){
					                    return confirm("Do you really want to delete \"" + element.filename + "\" ?");
					                });

					                drEvent.on('dropify.afterClear', function(event, element){
					                    alert('File deleted');
					                });
