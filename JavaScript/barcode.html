					 /*  				  生成条形码的插件
					 /*===============================================
							1、要引入的文件：
								1）、jquery.js
								2）、jquery-barcode.js
					
						   	2、使用：
								
								html:
								<input type="button" value="ean13" id="ean13Btn"></input>
								<input type="button" value="int25" id="int25Btn"></input>
								<div id="bcTarget"></div>
								<div id="bcTarget2"></div>
						   		
						   		js:
						   		$("#ean13Btn").click(function(){
									$("#bcTarget").barcode("1234567890128", "ean13", {
										barWidth : 2,
										barHeight : 30
									})
								})

								$("#int25Btn").click(function(){
									$("#bcTarget2").barcode(
										"1234567",
										"int25",
										{
											barWidth : 2 ,
											barHeight : 30
										}
									)
								})
