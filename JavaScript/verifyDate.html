 /*  	日期合法性验证
 /*===============================================*/
	
	function verifyDate( vDate ){
		/*
		验证格式必须是：  "YYYY-MM-DD hh:mm:ss"
		*/
		return /^(\d{4}-\d{2}-\d{2})\s{1}(\d{2}:\d{2}:\d{2})$/.test( vDate );
	}	
	
	alert( verifyDate( '2016-06-23 16:20:20' ) );

				
			
				
	代码说明：
	1、使用正则的test方法
	2、如果匹配，返回true, 否则返回false
