 /*  	获取指定日期是第几周
 /*===============================================*/
	 function getMonthDays( Y, M ){
		return new Date( Y, M, 0 ).getDate();
	}
	function getHowManyWeeks( Y, M, D ){
		var totalDays = 0;

		for( var i=0; i<M; i++ ){
			totalDays += this.getMonthDays( Y, M );
		}
		totalDays += D;
		return Math.ceil( totalDays/7 );
	}	
	

				
			
				
	代码说明：
	1、计算出M之前月份有多少天
	2、加上D天，即是所有累加的天数
	3、除以7，向上取整，算出指定日期是当年的第几周

	使用：
	console.log( getHowManyWeeks('2016','2','6') );
