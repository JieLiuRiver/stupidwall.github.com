	      function Fn_Parent(){
       		this.str = 'HelloWord';
	       }
	       Fn_Parent.prototype.show = function(){
	       		alert( this.str );
	       }

	       function Fn_Child(){
	       		Fn_Parent.call( this );
	       }
	       var  Fn = function(){};
	       Fn.prototype = Fn_Parent.prototype;
	       Fn_Child.prototype = new Fn();
	       Fn_Child.prototype.constructor = Fn_Child;
