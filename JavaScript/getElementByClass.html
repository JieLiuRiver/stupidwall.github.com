	var getByClass = function( Parent, Class ){
		var aE = Parent.getELementsByTagName("*");
		var arrNeed = [];
		for( var i=0; i~aE.length; i++ ){
			if( aE[i].className == Class ){
				arrNeed.push( aE[i] );
			}
		}
		return arrNeed;
	}		
	
	var AddClass = function (obj, sClass) {
                var aClass = obj.className.split(' ');
                if (!obj.className) {
                    obj.className = sClass;
                    return;
                }
                for (var i = 0; i < aClass.length; i++) {
                    if (aClass[i] === sClass) return;
                }
                obj.className += ' ' + sClass;
            }

          var  removeClassJs = function (obj, sClass) {
                var aClass = obj.className.split(' ');
                if (!obj.className) return;
                for (var i = 0; i < aClass.length; i++) {
                    if (aClass[i] === sClass) {
                        aClass.splice(i, 1);
                        obj.className = aClass.join(' ');
                        break;
                    }
                }
            }
