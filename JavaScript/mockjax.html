								//在发起请求之前写入：
								var isAjaxMocked = true;
								if( isAjaxMocked ){
									$.mockjax({
										url : 'testData.php',
										status : 200,
										responseTime : 750,
										response : function(){
											var data = $.mockJSON.generateFromTemplate({
												"list|20-30" : [
													{  
														"id|+1" : 1,

															}
														]  
													}

										}
