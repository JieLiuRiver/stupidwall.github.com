                    1)、属性callee    指的是当前正在执行的函数
                    可以通过arguments.callee来引用匿名函数自身
                    var fn1 = function(x){
                        if( x<2 ){
                            return 1;
                        }else{
                            return x * arguments.callee(x-1);
                        }
                    }

                    2)、Array
                        1)、concat()
                                合成新的数组
                                [1,2,3].concat(4,[5],[6,7]);

                        2)、every()
                                测试数组所有元素是否满足某些条件
                                [1,2,3].every(function(x){
                                    return x <2;
                                });  //false

                       3)、  filter()
                                [1,2,3].filter(function(x){
                                        return x > 1;
                                }) //  =>  [2,3]

                       4)、forEach()
                                [1,2,3].forEach(function(x,i,a){
                                        a[i]++;
                                }) //  =>  [2,3,4]

                       5)、indexOf()
                                ['a','b'].indexOf('b') //  =>  1

                       6)、join 转化为字符串
                                [1,2,3,'testing'].join("+");  //  "1+2+3+testing"

                       7)、 map()
                               [1,2,3].map(function(x){
                                    return x*x;
                               });   // =>  [1,4,9]

                       8)、pop()  删除并返回最后一个元素
                              shift()移除第一个
                              push()  追加到最后
                              unshift()添加到第一个

                       9)、reduce();  从数组元素中计算出一个值
                                var arr = [1,2,3,4,5];
                                var num = arr.reduce(function(x,y){
                                        return x*y;
                                });
                                alert( num );  // 120

                       10)、reduceRight() 从右到左
                                [2,10,60].reduceRight(function(x,y){
                                        return x/y;
                                })   //==> 3   (60/10)/2

                        11)、reverse() 颠倒数组中的元素
                                [2,10,60].reverse()  //[60,10,2]

                        12)、slice() 返回片段
                                [60,10,2,1].slice(0,3)  //[60,10,2]

                        13)、some() 测试是否有元素满足有些条件
                                [60,10,2,1].some(function(x){ return x >10 })  //true


                        14)、sort() 排序函数
                                [60,10,2,1].sort(function(a,b){ return a - b  }) //  1,2,10,60

                        15)、splice() 插入、删除、替换元素
                                    var arr = ['A','B','C','D']
                                    arr.splice(0  ,1)
                                    alert( arr );   //====> B C D
                                    arr.splice(0,2)   // ===>干掉2个
                                    //删除的返回值，是删除的对象

                                    arr.splice(0,1,'a')  //用a替换掉第一个A
                                    alert( arr )

                                    arr.splice(1,0,'e')
                                    alert( arr );   // =====> A e B C D

                             3)、instanceof ： 不是属性和方法， 而是运算符
                                    对象与构造函数在原型链上是否有关系
                                    function Aaa(){
                                    }
                                    var a1 = new Aaa();
                                    alert( a1 instanceof Aaa ); 查看实例对象a1与对象Aaa，在圆形链上是否有关系。如果是，返回true


                             4)、toString  完美的类型判断
                                var x =[] // {} // null // num // string //。。。。
                                alert( Object.prototype.toString.call(x ) );  // [object Array] // [object object] // [object number] // [object string]
                                可以这样判断：
                                if( Object.prototype.toString.call([]) == '[object Array]' ){
                                }

                            10)、substring()
                                var str = '妙味课堂是一只独具特色的IT培训团队'；
                                str.substring(4); // =======>  一只独具特色的IT培训团队
                                str.substring(0,2) // ====> 妙味
                                //这个substring就是截取部分
                                str.substring(2,0)// 这样的话====》妙味
                                //这玩意，可以检测里面的2个数字，自动调整数字的位置。
                                str.substring(); //截取全部，跟写个0 一样的
                                str.substring(-3,2); // -3当成0处理
