					4、js
						var arrImg=['img/biZhi_1.png','img/biZhi_2.png','img/biZhi_3.png','img/biZhi_4.png','img/biZhi_5.png','img/biZhi_6.png','img/biZhi_7.png','img/biZhi_8.png'];
						var _Len = arrImg.length;
						var oList = document.getElementById('list');
						var oBox = document.getElementById('box');
						var aBtn = oBox.getElementsByTagName('a');
						var aLi = null;
						var iNow = 0;
						var zIndex = _Len;
						loadImg();
						function loadImg(){
							var num = 0;
							for( var i=0; i<_Len; i++ ){
								var Img = new Image();
								Img.src=arrImg[i];
								Img.onload = function(){
									num++;
									if( num == _Len ){
										showImg();
									}
								}
							}
						}
						
						//设置第一个层级最高 最后一个层级最低
						function showImg(){
							var aHtml = '';
							arrImg.forEach(function(v,i){
								aHtml += "<li style='background:url("+ v +") no-repeat;background-size:cover;z-Index:"+ (_Len - i) +"'></li>";
							})
							oList.innerHTML = aHtml;
							
							aLi=oList.children;
							for( var i=0; i<aLi.length; i++ ){
								var iDeg = parseInt( Math.random()*100 )%5 + i*5;
								aLi[i].iDeg = iDeg;
								aLi[i].style.WebkitTransform = "rotate("+ iDeg +"deg) scale(1.5)";
							}
							setTimeout(function(){
								show();
							},300)
						}
						function show(){
							for( var i=0; i<aLi.length; i++ ){
								aLi[i].style.transition = ".5s "+ (aLi.length - i)*400 +"ms";
								aLi[i].style.WebkitTransform = "rotate("+ aLi[i].iDeg +"deg) scale(1)";
								aLi[i].style.opacity = 1;
							}
						}
						
						aBtn[0].onclick = fnPrev;
						function fnPrev(){
							iNow--;
							if( iNow < 0 ){
								iNow = aLi.length-1;
							}
							aLi[iNow].addEventListener('webkitTransitionEnd',endPrev, false);
							aLi[iNow].style.transition = '1s'; 
							aLi[iNow].style.left = '-100%';
							aLi[iNow].style.opacity = 0;
						}
						function endPrev(){
							var that = this;
							zIndex++;
							this.style.zIndex = zIndex;
							this.removeEventListener('webkitTransitionEnd',endPrev, false);
							this.style.transition = 'none';
							this.style.WebkitTransform = "rotate("+ this.iDeg +"deg) scale(1.2)";
							setTimeout(function(){
								that.style.transition = '.6s';
								that.style.opacity = 1;
								that.style.left = 0;
								that.style.WebkitTransform = "rotate("+ that.iDeg +"deg) scale(1)";
							},50);
						}
						
						aBtn[1].onclick = fnNext;
						function fnNext(){
							iNow++;
							if( iNow > aLi.length-1 ){
								iNow = 0;
							}
							aLi[iNow].addEventListener('webkitTransitionEnd',endNext, false);
							aLi[iNow].style.transition = '1s'; 
							aLi[iNow].style.left = '100%';
							aLi[iNow].style.opacity = 0;
						}
						function endNext(){
							var that = this;
							zIndex++;
							this.style.zIndex = zIndex;
							this.removeEventListener('webkitTransitionEnd',endNext, false);
							this.style.transition = 'none';
							this.style.WebkitTransform = "rotate("+ this.iDeg +"deg) scale(1.2)";
							setTimeout(function(){
								that.style.transition = '.6s';
								that.style.opacity = 1;
								that.style.left = 0;
								that.style.WebkitTransform = "rotate("+ that.iDeg +"deg) scale(1)";
							},50);
						}
