				 /*  	禁止滚动条
				 /*===============================================*/
					1、css实现的方法：
						body{ overflow-x:hidden }
						
					2、js实现：
						document.onmousewheel = function(event){
							var event = event || window.event;
							if( event && event.preventDefault ){
								event.preventDefault();
								event.stopPropagation();
							}else{
								return false;
							}
						}
