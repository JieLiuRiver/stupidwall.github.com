        		if( e.addEventListener ){
        			e.addEventListener(type,fn,false);
        		}else if(e.attachEvent){
        			e.attachEvent('on'+type,function(){
        				fn.call(e);
        			});
        		}else{
        			e['on'+type] = fn;
        		}
        	},
        	remove : function(e,type,fn){ 
        		if( e.removeEventListener ){

        				fn.call(e);
        			})
        		}else{ 
        			e['on'+type] = null;	
        		}
