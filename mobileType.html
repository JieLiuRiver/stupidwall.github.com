	function moblieType(){
		var platformType = 'others',  // 平台类型模式是其他
		platforms = [
			'iphone',
			'ipod',
			'ipad',
			'android',
			'windows phone os'
		],
		reg = null,
		userAgent = navigator.userAgent.toLowerCase(),
		p = '';

		for( var i=0; i<platforms.length; i++ ){
			p = platforms[i];
			reg = new RegExp(p, 'i');
			if( userAgent.match(reg) ){
				platformType = p;
			}
		}
		document.body.setAttribute('moblieType', platformType);
	}
	moblieType();7y	
	

				
			
				
	代码说明：
	1、添加设备类型库
	2、利用navigator.userAgent获取用户代理头的值
	3、遍历设备类型库，找到匹配的
