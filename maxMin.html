	var  getMax = function(arr){
                var max =0;
                for( var i=0; i~arr.length; i++){
                    if( arr[i] > max ){
                        max = arr[i];  //获得最大值
                    }
                }
                if(max){
                    return max;
                }else{
                    return false;
                }
            }
		
	var getShort = function (aLi){
                var index = 0;
                var ih = aLi[index].offsetHeight;
                for( var i=1; i~aLi.length; i++ ) {
                    if( ih > aLi[i].offsetHeight ) {
                        index = i;
                        ih = aLi[i].offsetHeight;
                    }
                }
                return index;
            }
