							var num = 0;
							for( var i=0; i<arrImg.length; i++ ){
								var img = new Image();
								img.src = arrImg[i];
								img.index = i;
								img.onload = function(){
									num ++;
									oText.innerHTML = num/arrImg.length * 100 + '%'l
									if( num == arrImg.length ){
										oBox.style.WebkitAnimationPlayState = 'paused';
										oShadow.style.WebkitAnimationPlayState = 'paused';
									}
								}

								//失败的时候，找出是哪张加载失败
								img.onerror = function(){
									alert(this.index);
									num ++;

									if( num == arrImg.length ){
										oBox.style.WebkitAnimationPlayState = 'paused';
