													function id(id){
														return document.getElementById(id);
													}
													var oList = id('list');
													var aLi = oList.children;
													var oBtnWrap = id('btnWrap');
													var aLi_b = oBtnWrap.children;
													
													touchBtn();
													function touchBtn(){
															for( var i=0; i<aLi_b.length; i++ ){
																	aLi_b[i].index = i;
																	aLi_b[i].onmouseover  = function(){
																			for( var k=0; k<aLi_b.length; k++ ){
																					aLi_b[k].style.left = '-80px';
																					aLi_b[k].style.background = '#f60';
																			}
																			this.style.left = 0;
																			this.style.background = '#333';
																	}
															}
													}
													
													clickBtn();
													function clickBtn(){
															for( var i=0; i<aLi_b.length; i++ ){
																	var e = aLi_b[i];
																	e.onclick = function(){
																			for( var i=0; i<aLi.length; i++ ){
																					aLi[i].index = i;
																			}
																			for(  var i=0; i<aLi.length; i++ ){
																					if( this.index ==  aLi[i].index){
																							switch( i+1 ){
																									case 4 : 
																										oList.style.WebkitTransform = 'rotateY(90deg)';
																									break;
																									case 2 : 
																										oList.style.WebkitTransform = 'rotateY(-90deg)';
																									break;
																									case 1 : 
																										oList.style.WebkitTransform = 'rotateY(0deg)';
																									break;
																									case 3 :
