						//怎么判断 鼠标移到的是 一个元素的左侧还是右侧呢？
						oSpan.onmouseover = fnOver;
						function fnOver ( ev ){
							var cliX = ev.clientX;
							var offsetLeft = getPos(this).left;
							var l = clienX - offsetLeft;
							this.style.transition = '.5s';
							if( l > this.offsetWidth/2 ){
								console.log("R");
								oSpan.style.WebkitTransform = 'rotateY(30deg)';
							}else{
								console.log("L");
								oSpan.style.WebkitTransform = 'rotateY(-40deg)';
							}
						}
						function getPos( obj ){
							var pos = 	{ left:0,top:0 }
							while( obj ){
								pos.left += obj.offsetLeft;
								pos.top += obj.offsetTop;
								obj = obj.offsetParent;
							}
							return pos;
						}
					}
