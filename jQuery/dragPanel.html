					 /*  				  	可拖拽和最大化最小化的多功能浮动面板jQuery插件
					 /*===================================================================
							1、要引入的文件：
								1）、jquery-ui.complete.min.css
								2）、jquery.jspanel.css
								3）、js/jquery-2.1.1.min.js
								4）、js/jquery-ui-complete.min.js
								5）、js/jquery.ui.touch-punch.min.js
								6）、js/jquery.jspanel.min.js
					
						   	2、使用：
								html：
								<button id="btn1">点击按钮</button>
								

								js：
								//基本用法
								$('#btn1').click(function(){
								            $.jsPanel({
								            		position: {
										        left: 200,
										        top:  100
										    },
										    title:    'jsPanel theme info',
										    theme:    'info'
								            });
								 });
								
								//特殊用法
								$.jsPanel({
								    selector:  '.testcontainer',
								    theme:     'dark',
								    position:  'center',
								    draggable: {
								        containment: "parent"
								    }
								});
