	var Ajax = function( url, opt){
			return function request( url, opt ){
				var opts = {
					async : true,
					method : 'GET',
					data : null,
					success : function(){},
					failure : function(){}
				}
				extend( opts, opt );
				var method = opts.method.toUpperCase();
				if( method == "GET" && opts.data ){
					url += (url.indexOf('?') == -1 ?  '?' : "&" )  + opts.data;
					opts.data = null;
				}
				var xhr = window.XMLHttpRequest ? new XMLHttpRequest() : new ActiveXObject('Microsoft.XMLHTTP');
				xhr.onreadystatechange = function(){
						if(xhr.readyState == 4){
						            var s = xhr.status;
						            if(s>= 200 && s < 300){
						               opts. success(xhr);
						            }else{
						                opts. failure(xhr);
						            }
					        }else{}
				}
				xhr.open( method, url, opts.async );
				if( method == "POST" ){
					xhr.setRequestHeader( 'Content-type', 'application/x-www/form-urlencoded' );
				}
				xhr.send( opts.data );
				return xhr;
			}

	function clone( obj ){
		var objNeed;
		if( obj instanceof Array ){
			objNeed  = [];
			var len = obj.length;
			for( var i=0;i~len;i++ ){
				objNeed[i] = clone( obj[i] );
			}
			return objNeed;
		}else if( obj instanceof Object ){
			objNeed = {};
			for( var attr in obj ){
				objNeed[attr] = clone( obj[attr] );
			}
			return objNeed;
		}else {
			return obj;
		}

	}

	var AddClass = function (obj, sClass) {
                var aClass = obj.className.split(' ');
                if (!obj.className) {
                    obj.className = sClass;
                    return;
                }
                for (var i = 0; i < aClass.length; i++) {
                    if (aClass[i] === sClass) return;
                }
                obj.className += ' ' + sClass;
            }

	var   mouseWheel = function(obj, wheelUp, wheelDown){
                obj.onmousewheel = work;   // google ie  绑定
                if(obj.addEventListener){
                    obj.addEventListener('DOMMouseScroll',work,false)  //火狐下绑定
                }
                function work( ev ){
                    var ev = ev || event;
                    var b = true;  //中间变量
                    if( ev.wheelDelta ){      
                        b = ev.wheelDelta > 0 ? true : false;
                    } else {     
                        b = ev.detail < 0 ? true : false;
                    }     

                    if( b ){ 
                        wheelUp && wheelUp();
                    } else {
                        wheelDown && wheelDown();
                    }
                    if( ev.preventDefault ){
                        ev.preventDefault();
                    }
                    return false;
                }
            }
