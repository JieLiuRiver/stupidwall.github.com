 /*  	移除移动浏览器地址栏
 /*===============================================*/
		
	window.onload = function(){

		var orientationCall = [];    //函数队列

		var orientation = (function(){
			window.addEventListener( 'orientationchange', function( event ){
				orientation.forEach( function(value, index){
					value();
				} );
			} );
		})();

		function addOrientation( callFn ){    //添加回调函数
			orientationCall[ orientationCall.length ] = callFn;
		}

		function clearUrl(){
			var scroll = function(){
				window.scrollTo(0, 1);
			}
			scroll();
			addOrientation( scroll );
		}
	}

				
			
				
	代码说明：
	1、 当页面内容过多，超过屏幕高度时，自动隐藏地址栏
	2、window.scroll(0, 1);  核心代码
