	                       1)、依赖的文件
                                1)、jquery.min.js
                                2)、bootstrap.js
                                3)、angular.js

                                8)、animation.css
                            
                            2)、html
                                <input type="button" class="btn btn-primary" value="弹出" ng-click="modal()" />

                            3)、 js
                            
                                angular.module('myApp',['mgcrea.ngStrap'])
                                    .controller('myCtrl',['$scope','$alert','$modal',function($scope,$modal,$alert){
                                         $scope.modal = function(){
                                            $modal({
                                                title : '提示',
                                                content : 'Hello World',
                                                template :　"modal/modal.confirm.tpl.html",
                                                animation : "a-flipinY",
                                                callback : function(){
                                                    alert('ok');
                                                }
                                            });
                                         }
                                    }]);
