 /*  	指令之间的通信
 /*===============================================*/
	<hello>
		<hi></hi>
	</hello>

=========================
	var m1 = angular.module('myApp',[]);
	m1.directive('hello',function(){
		return {
			restrict : 'E',   
			replace : true,
			transclude : true,
			controller : function($scope){   //conroller更多的时数据模型
				//$scope.name = 'miaov';
				this.name = 'miaov';    // 这里用this，方便后面属性传递到别的指令
				this.show = function(){};
			},
			template : '<div>hello angular<h1 ng-transclude></h1></div>'
		};
	});
	// ng-transclude :  意味着hi指令的内容会放在我们响应的 h1标签里去，
	//不写的话，会覆盖掉hello内容


	m1.directive('hi',function(){
		return {
			restrict : 'E',   
			replace : true,
			require : '?^hello',   //意味着接收hello指令，并且往上找^，如果自身找不到hello，用来?不会报错
			template : 'hi angular',
			link : function(scope,element,attr,reController){    //link更多的是dom操作
				console.log( reController );
			}
		};
	});
