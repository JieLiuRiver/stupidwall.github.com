						<h3>{ { data | filter : argument1 : argument2 : ....  } }</h3>
						<span ng-repeat="a in array | filter"></span>
						
						1)、{ { num | currency : "￥" } }  //货币处理
						2)、{ { num | date : "yyyy-MM-dd hh:mm:ss EEEE" } }  //日期格式化  E表示星期

						3)、还有一个filter 用来匹配子串 容易混淆 用来处理数组 过滤出某个子串的元素
						$scope.array = [
							{ name : "dsfd", age : 3 },
							{ name : "sefe", age : 4 },
							{ name : "dgh5h", age : 5 },
							{ name : "ygyh", age : 6 },
							{ name : "rghrh", age : 7 },
							{ name : "drfgdr", age : 8 }
						]
						$scope.fn = function(e){
							return e.age > 4
						}

						{ { array | filter : 'a' } } //匹配属性值中含有a的
						{ { array | filter : 4 } }  //属性值中含有4的匹配出来
						{ { array | filter : { name : 'r' } } } //参数是对象， 匹配name属性中含有r的
						{ { array | filter : fn } } //一个函数，匹配函数里的条件 
						
						4)、json------用来把一个js对象格式化为一个json字符串
						可以用来调试
						{ { jsonTest | json } }

						5)、 limitTo -----用来截取数组或者字符串 
						{ { array | limitTo : 2 } } //将显示前两项

						6)、lowercase  uppercase  

						7)、orderBy  排序
							{ { array | orderBy : 'age' } }  //按age属性值进行排序， 若是-age则是倒序
							{ { array | orderBy : orderFn } } //按照函数的返回值进行排序
							{ { array | orderBy : ['age','name'] } } //如果age相同，就按照name进行排序

						8)、自定义过滤器
						app.filter('odd',function(){
							return function(inputArray){
								var array = [];
								for( var i=0; i<inputArray.length; i++ ){
										if( i%2 != 0 ){
											array.push(inputArray[i]);
										}
								}
								return array;
							}
						});
