		//这里的this指针指向的是$scope.vm变量，作为一个约定，我们把它赋值给vm变量，不再使用this
		var vm = this;
		$scope.submit = function(form){

		}
		})
		.directive('invalidError',function( $compile ){
		return{
		restrict : 'A',
		require : 'ngModel',
		link : function(scope,element,attrs,ngModel){
		console.log('wrong?');
		var subScope = scope.$new(true);
		subScope.hasError = function(){
		return ngModel.$dirty && ngModel.$invalid;
		}
		subScope.errors = function(){
		return ngModel.$error;
		}
		var hint = $compile('<ul class="error" ng-if="hasError()">
				<<li ng-repeat="(name, wrong) in errors()" ng-if="wrong">
				{ {name | error} }</li></ul>')(subScope);

		element.after(hint);
		}
		}
		});
