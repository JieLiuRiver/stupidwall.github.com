 /*  	省市区的插件的使用
 /*===============================================*/
        1、需要引入的文件：
        	1)、jquery.js
        	2)、bootstrap.min.css
        	3)、bootstrap.min.js
        	4)、pca-selector.js

		2、在pca-selector.js里引入的文件：
			1)、pca.json
			2)、box.html

		3、HTML上应用：
			<input  
                    select-address 
                    name="address" 
                    ng-model="formData.address" 
                    provinceid="formData.provinceid"
                    cityid="formData.cityid" 
                    areaid="formData.areaid" 
                    provincename="formData.provincename" 
                    cityname="formData.cityname" 
                    areaname="formData.areaname" 
                    placeholder="请选择所在地" 
                    type="text" 
                    class="form-control pca-input" 
                    style="width: 200px;"/>

		4、在box.html里引入里一个style.css文件

		5、controller写法：

			//注入指令模块
			app = angular.module('app', ['util.selectAddress']);
			
			//设置默认的省市区
			app.controller('pcaController', ["$scope",function ($scope) {
				$scope.formData={};
			    $scope.formData.provinceid = 19;
			    $scope.formData.cityid = 197;
			    $scope.formData.areaid = 1752;
			}]);
