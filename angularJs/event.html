 /*  	事件内部的传播方式
 /*===============================================*/
  	var m1 = angular.module('myApp',[]);
	m1.controller('Aaa',['$scope',function($scope){	
		$scope.count = 0;
		$scope.$on('myEvent',function(event){		
			$scope.count++;		
			//console.log( event.targetScope == event.currentScope );		
			//console.log(event.name);		
			event.stopPropagation();		
		});
	}])
	</script>
	</head>
	<body>
	<div ng-controller="Aaa">
		{{count}}
	    <div ng-controller="Aaa" ng-click="$emit('myEvent')">
	        {{count}}
	        <div ng-controller="Aaa">
	            {{count}}
	        </div>
	    </div>
	</div>
	</body>


				
			
				
	代码说明：
	1、我们的控制器，进行内嵌
	2、然后中间的变量通过事件改变，
	     会影响父级和子级的吗？
	3、默认是不会的
	4、监听事件 $scope.$on('event',function(event){})
	5、向上传播方式绑定事件： ng-click="$emit(event)"
	6、向下传播方式绑定事件： ng-click="$broadcast(event)"
	7、时间对象：event
		targetScope  传播到目标上对应的作用域
		currentScope 当前的作用域
