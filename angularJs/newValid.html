						var m1 = angular.module('myApp',[]);
						m1.controller('Aaa',['$scope',function($scope){	
							$scope.regText = {
								regVal : 'default',
								regList : [
									{ name : 'default' , tips : '请输入用户名'},
									{ name : 'required' , tips : '用户名不能为空'},
									{ name : 'pattern' , tips : '用户名必须是字母'},
									{ name : 'pass' , tips : '√'}
								]
							};
							
							$scope.regPassword = {
								regVal : 'default',
								regList : [
									{ name : 'default' , tips : '请输入密码'},
									{ name : 'required' , tips : '密码不能为空'},
									{ name : 'minlength' , tips : '密码至少6位'},
									{ name : 'pass' , tips : '√'}
								]
							};
							
							$scope.change = function( reg , err ){
								for(var attr in err){
									if( err[attr] == true ){
										$scope[reg].regVal = attr;
										return;
									}
								}
								$scope[reg].regVal = 'pass';
							};
						}]);

					app.controller('ctrl',function($scope){
						$scope.getClass = function(model){
							return {
								error : model.$invalid && model.$dirty,
								success : model.$valid && model.$dirty
							}
						}
						$scope.showError = function(model, error){
							return model.$error[error];
						}
					});
				
					//让保存按钮无效
					<form name="form">
						<button ng-disabled="!canSave()"></button>
					</form>

					$scope.canSave = function(){
						return $scope.form.$dirty && $scope.form.$valid;
					}
