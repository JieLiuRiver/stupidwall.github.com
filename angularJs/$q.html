 /*  	$q延迟的使用
 /*===============================================*/
        var m1 = angular.module('myApp',[]);
            m1.controller('Aaa',['$scope','$q',function($scope,$q){
            var dfd = $q.defer();
            function show(){
                setTimeout(function(){
                dfd.reject();
            },2000);
             return dfd.promise;
        }

        show().then(function(){
            alert('成功');
            },function(){
            alert('失败');
            });
        }]);

				
			
				
	代码说明：
    1、创建延迟对象$q.defer();
    2、resolve()表示成功
    3、reject()失败状态
