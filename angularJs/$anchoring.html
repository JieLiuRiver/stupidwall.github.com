    <script src="angular.min.js"></script>
    <script src="angular-animate.js"></script>
    <script src="angular-route.js"></script>

    <h1><a href = "#/">主页</a></h1>
    <div class="view-container">
    <div class="view" ng-view></div>
    </div>

    <script>
    angular.module('MyApp',['ngAnimate','ngRoute'])
    .config(['$routeProvider', function($routerProvider){
    $routerProvider.when('/',{
    templateUrl : 'home.html',
    controller : 'HomeController'
    });
    $routerProvider.when('/item/:id',{
    templateUrl : 'item.html',
    controller : 'ItemController'
    });
    }])
    .run(['$rootScope', function($rootScope){
    $rootScope.items = [
    { id : '1', item : 'item1' },
    { id : '2', item : 'item2' },
    { id : '3', item : 'item3' },
    { id : '4', item : 'item4' },
    { id : '5', item : 'item5' }
    ];
    }])
    .controller('HomeController',['$scope','$rootScope',function($scope,$rootScope){

    }])
    .controller('ItemController',['$scope','$rootScope','$routeParams',function($scope,$routeParams,$rootScope){
    }])
    </script>


    home.html:
    <h1>欢迎来到我的主页</h1>
    <p>点击下面的元素</p>
    <a ng-repeat="i in items" ng-href="#/item/{ { i.id } }" class="item" >
    { { i.item } }
    </a>


				
			
				
	代码说明：
    1、ng-view
    2、注入
    3、通过run方法
    4、绑定对应的控制器
    5、$routeParams里去找对应的id
