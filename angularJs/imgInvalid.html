 /*  	自定义图形验证指令
 /*===============================================*/
	app .directive('bfCaptcha', function bfCaptcha(){
		return {
			restrict : 'A',
			link : function(scope, element){
			var changeSrc = function(){
			element.attr('src','api/captcha.jpg?random='+new Date().getTime());
			}
			changeSrc();
			element.on('click', function(){
			changeSrc();
			});
		}
	}
	})


				
			
				
	代码说明：
	使用：<img bf-captcha alt="图形验证码"/>
