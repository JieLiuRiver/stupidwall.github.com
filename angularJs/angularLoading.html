/*===============================================*/
												 /*  					angularLoading插件使用
												 /*===============================================*/
       													1、需要引入的文件：
       														1)、angular.js
       														2)、loading-bar.min.js	

       													2、使用：
       														<button ng-click="tt()">点击按键</button>		

       													3、js:
       														angular.module('myApp',['chieffancypants.loadingBar'])
																.controller('testCtrl',function($scope,$http){
																		var config = {
																				url :'test.php',
																				method : 'get'
																		}
																		$http(config).success(function(data){
																			console.log( data );
																		})
																		$scope.tt = function(){
														                    $http(config).success(function(data){
														                   	 	console.log(data);
													                		})
														                }
																});
