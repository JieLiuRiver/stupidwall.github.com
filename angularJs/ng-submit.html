 /*  	ng-submit 
 /*===============================================*/
	
	<form ng-submit="myFunc()">
	    <input type="text">
	    <input type="submit">
	</form>

	<p>{ {myTxt} }</p>

	<script>
	var app = angular.module("myApp", []);
	app.controller("myCtrl", function($scope) {
	    $scope.myTxt = "你还没有点击提交!";
	    $scope.myFunc = function () {
	        $scope.myTxt = "你点击了提交!";
	    }
	});
	</script>
	

				
			
				
	代码说明：
	ng-submit 指令用于在表单提交后执行指定函数。
