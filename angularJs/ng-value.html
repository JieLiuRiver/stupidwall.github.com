 /*  	ng-value 
 /*===============================================*/
	<div ng-app="myApp" ng-controller="myCtrl">

	<input ng-value="myVar">

	</div>

	<script>
	var app = angular.module('myApp', []);
	app.controller('myCtrl', function($scope) {
	    $scope.myVar = "Hello World!";
	});
	</script>
	
	

				
			
				
	代码说明：
	ng-value 指令用于设置 input 或 select 元素的 value 属性。
