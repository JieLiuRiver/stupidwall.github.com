/*===============================================*/
												 /*  					decimal插件使用
												 /*===============================================*/
								1、需要引入的文件：
									1)、angular.js
									2)、decimal.js	

								2、使用：
									<input type="text" ng-model="text1" decimal>-->可输入<br>
							   	<input type="text" ng-model="text2" ng-disabled="true" decimal>-->不可用<br>
							   	<input type="text" ng-model="text3" ng-readonly="true" decimal>-->只可读<br>
							   	<input type="text" ng-model="text4" decimal precision="1">-->可输入(保留1位小数)<br>
							   	<input type="text" ng-model="text5" decimal precision="3">-->可输入(保留3位小数)<br>		

								3、js:
									var app=angular.module("app",["util.decimal"]);
								app.controller("decimalController",["$scope",function($scope){
									//初始化
									$scope.text1=0; //可输入
									$scope.text2=12; //不可用
									$scope.text3=50; //只读
									$scope.text4=0; //可输入(保留1位小数)
									$scope.text5=0; //可输入(保留3位小数)
								}]);

							4、
								decimal 小数指令--必填项
									precision 精度值--可选项，默认为2
