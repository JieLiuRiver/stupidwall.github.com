 /*  	倒计时
 /*===============================================*/
	function getCountDown(Y, M, D, h, m, s){
		Y = Y || 0;
		M = M || 0;
		D = D || 0;
		h = h || 0;
		m = m || 0;
		s = s || 0;
		var date = new Date( Y, M-1, D, h, m, s );

		//转换为时间戳
		var times = date.getTime() - new Date().getTime();
		return Math.ceil( times/(1000*60*60*24) );  //返回天数 
	}
	

				
			
				
	代码说明：
	1、计算距离指定日期的天数，
	2、要知道当前日期，指定日期
	3、日期转换为时间戳
	
	使用：
	console.log( getCountDown('2016','2','14') );
